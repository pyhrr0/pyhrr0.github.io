Rust: monomorphisation

use std::ops::Add;

fn add<T: Add<Output = T>>(a: impl Into<T>, b: impl Into<T>) -> T {
    a.into() + b.into()
}

fn main() {
    add::<i64>(42_i64, 42_i32);
    add::<i32>(42_i32, 42_i16);
    add::<i16>(42_i16, 42_i8);
    add::<i8>(42_i8, 42_i8);
    add::<f64>(42, 4.2);
    add::<f64>(4.2, 42);
}
